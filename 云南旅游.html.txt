    <script>
        // Check system
        let completedTasks = new Set();
        
        function toggleCheck(btn, id) {
            btn.classList.toggle('completed');
            if (completedTasks.has(id)) {
                completedTasks.delete(id);
            } else {
                completedTasks.add(id);
                // Simulate partner notification
                setTimeout(() => {
                    showNotification('同伴完成了 "' + btn.nextElementSibling.querySelector('h3').innerText + '"');
                }, 1000);
            }
            updateProgress();
        }

        function updateProgress() {
            const total = 20;
            const current = completedTasks.size;
            const percent = (current / total) * 100;
            document.querySelector('.bg-gradient-to-r.from-blue-500').style.width = percent + '%';
            document.querySelector('.text-xs.text-gray-400').textContent = `已完成 ${current}/${total} 项`;
        }

        function updateChecklist(checkbox) {
            if (checkbox.checked) {
                showNotification('已添加到完成清单');
            }
        }

        // Chat system
        function toggleChat() {
            const modal = document.getElementById('chat-modal');
            modal.classList.toggle('hidden');
            if (!modal.classList.contains('hidden')) {
                document.getElementById('chat-badge').classList.add('hidden');
                scrollToBottom();
            }
        }

        function sendMessage() {
            const input = document.getElementById('chat-input');
            const text = input.value.trim();
            if (!text) return;

            // Add user message
            addMessage(text, 'me');
            input.value = '';
            scrollToBottom();

            // Simulate partner typing and response
            showTyping();
            setTimeout(() => {
                hideTyping();
                const responses = [
                    "好的，听你的！",
                    "那我们要不要带件厚外套？",
                    "我刚刚查了一下，那家店评分很高",
                    "记得提前买票哦",
                    "哈哈，期待！"
                ];
                addMessage(responses[Math.floor(Math.random() * responses.length)], 'partner');
                if (document.getElementById('chat-modal').classList.contains('hidden')) {
                    document.getElementById('chat-badge').classList.remove('hidden');
                }
            }, 2000);
        }

        function addMessage(text, sender) {
            const container = document.getElementById('chat-messages');
            const div = document.createElement('div');
            div.className = 'chat-bubble flex gap-2 ' + (sender === 'me' ? 'flex-row-reverse' : '');
            
            const avatar = sender === 'me' ? 
                '<div class="w-8 h-8 bg-blue-500 rounded-full flex-shrink-0 flex items-center justify-center text-white text-xs">我</div>' :
                '<div class="w-8 h-8 bg-pink-500 rounded-full flex-shrink-0 flex items-center justify-center text-white text-xs">TA</div>';
            
            const bubbleClass = sender === 'me' ? 
                'bg-blue-500 text-white rounded-tr-none' : 
                'bg-white rounded-tl-none';
            
            const timeClass = sender === 'me' ? 'text-blue-200' : 'text-gray-400';
            
            div.innerHTML = `
                ${avatar}
                <div class="${bubbleClass} p-3 rounded-2xl shadow-sm max-w-[80%]">
                    <p class="text-sm">${text}</p>
                    <span class="text-xs ${timeClass} mt-1 block">${new Date().toLocaleTimeString('zh-CN', {hour: '2-digit', minute:'2-digit'})}</span>
                </div>
            `;
            
            container.appendChild(div);
            scrollToBottom();
        }

        function showTyping() {
            document.getElementById('typing-indicator').classList.remove('hidden');
            scrollToBottom();
        }

        function hideTyping() {
            document.getElementById('typing-indicator').classList.add('hidden');
        }

        function scrollToBottom() {
            const container = document.getElementById('chat-messages');
            container.scrollTop = container.scrollHeight;
        }

        function showNotification(text) {
            // Simple notification simulation
            const div = document.createElement('div');
            div.className = 'fixed top-20 right-4 bg-gray-800 text-white px-4 py-2 rounded-lg shadow-lg text-sm z-50 animate-bounce';
            div.textContent = text;
            document.body.appendChild(div);
            setTimeout(() => div.remove(), 3000);
        }

        // Budget modal
        function toggleDetailBudget() {
            document.getElementById('budget-modal').classList.toggle('hidden');
        }

        // Location info for map
        function showLocation(loc) {
            // Implementation for map interaction
        }
